                        const stack = [];
                        let openingIndex;
                        let closingIndex;
                        for (const item of string) {
                            if (openingSymbols.includes(item)) {
                                stack.push(item);

                                openingIndex = openingSymbols.indexOf(stack.pop());
                                if (openingIndex !== closingIndex) {
                                    return false;
                                }
                            }
                        }
                        return stack.length === 0;
